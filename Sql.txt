CREATE TABLE `Employee` (
	`ID` INT(10) NOT NULL AUTO_INCREMENT,
	`Name` VARCHAR(200) NULL DEFAULT NULL,
	`Email` VARCHAR(200) NULL DEFAULT NULL,
	`Phone` VARCHAR(200) NULL DEFAULT NULL,
	`employeeCode` VARCHAR(200) NULL DEFAULT NULL,
	PRIMARY K EY (`ID`)
)
COLLATE='utf8mb4_general_ci'
ENGINE=InnoDB
AUTO_INCREMENT=1
;


CREATE TABLE `Projetcs` (
	`ID` INT(10) NOT NULL AUTO_INCREMENT,
	`projectName` VARCHAR(200) NULL DEFAULT NULL,
	`NumberEmp` VARCHAR(200) NULL DEFAULT NULL,
	PRIMARY KEY (`ID`)
)
COLLATE='utf8mb4_general_ci'
ENGINE=InnoDB
AUTO_INCREMENT=1
;
